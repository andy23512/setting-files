# repo for setting files I using

## VIM
### File
vimrc
### Setup
$ cp vimrc ~/.vimrc
$ mkdir -p ~/.vim/bundle 
$ git clone https://github.com/VundleVim/Vundle.vim.git ~/.vim/bundle/Vundle.vim
$ vi
(in vi) :BundleInstall

## Git
gitconfig
### Setup
$ cp gitconfig ~/.gitconfig

## tmux
tmux.conf
### Setup
$ cp tmux.conf ~/.tmux.conf

## Bash
bashrc
### Setup
$ cp bashrc ~/.bashrc
